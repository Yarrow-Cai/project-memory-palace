import "sort"

// Sorted slices for stable MCP schema enum generation.
var MemoryTypeEnum = sortedKeys(MemoryTypes)
var SourceKindEnum = sortedKeys(SourceKinds)
var MemoryStatusEnum = sortedKeys(MemoryStatuses)

func sortedKeys(m map[string]bool) []string {
    keys := make([]string, 0, len(m))
    for k := range m { keys = append(keys, k) }
    sort.Strings(keys)
    return keys
}
